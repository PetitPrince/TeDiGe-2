  2 /** Encode a string with deflate
  3 
  4 	@param {string} Input string
  5 	@return {string} Output encoded string
  6 */
  7 Diagram.prototype.flate_encode = function (str) {
  8 return $.base64.encode(RawDeflate.deflate(unescape(encodeURIComponent(str))));
  9 };

 13 	@param {string} Input string
 14 	@return {string} Output decoded string

 16 Diagram.prototype.flate_decode = function(str) {
 17 return decodeURIComponent(escape(RawDeflate.inflate($.base64.decode(str))));
